175  int ret = 0;
176  bool frameisonset;
177  //printf("frames at %f size %d num %d\n", time, size, num);
178 
179  // clip colindex/size
180  //TODO: this shouldn't change at runtime, so do this in streamAttributes only
181  while (colindex < 0 && size > 0)
182  colindex += size;
183 
184  if (numcols <= 0)
185  numcols = size;
186 
187  if (colindex + numcols > (int) size)
188  numcols = size - colindex;
189 
190  for (unsigned int i = 0; i < num; i++)
191  { // for all frames
192  /* input frame */
193  double energy = 0.0;
194  unsigned int k = colindex;
195  for (int j = 0; j < numcols; j++, k++)
196  {
197  energy += values[k];
198  }
199  energy /= numcols;
200 
201  /* determine if there is an onset */
202  if (this->segison)
203  // within segment, check for max size if given
204  frameisonset = maxsize > 0 && time >= this->onsettime + maxsize;
205  else
206  // within silence, check for onset (but avoid re-trigger)
207  frameisonset = energy > onsetThreshold && time >= this->onsettime + minimumInterval;
208 
209  if (!this->reportduration)
210  { /* output marker only */
211  if (frameisonset)
212  { /* report immediate onset */
213  ret = this->propagateFrames(this->offset + time, weight, NULL, 0, 1);
214  this->onsettime = time;
215  }
216  }
217  else
218  { // check for onset and offset (segment begin and end)
219  double duration = time - this->onsettime;
220 
221  // check for segment end
222  if (this->segison && ((energy < offThreshold && duration >= durationThreshold)
223  || (maxsize > 0 && time >= this->onsettime + maxsize)))
224  { // energy below off threshold or max segment size exceeded
225  int outputsize = static_cast<int>(this->outputvalues.size());
226 
227  this->outputvalues[0] = duration;
228 
229  /* get temporal modelling */

232 
233  /* report segment */
234  ret = this->propagateFrames(this->offset + this->onsettime, weight, &this->outputvalues[0], outputsize, 1);
235  }
236

238  if (frameisonset)

242  }
