95 static double
96 getItur468Factor(double freq)
97 {
98  int index = getClosestItur468Index(freq);
99  double levl = 0.0;
100 
101  if(index < 0)
102  {
103  double freq0 = itur468Coeffs[0][0];
104  double levl0 = itur468Coeffs[0][1];
105 
106  levl = levl0 + log2(freq / freq0) * 6.0;
107  }
108  else
109  {

111  double freq1 = itur468Coeffs[index + 1][0];

113  double levl1 = itur468Coeffs[index + 1][1];

116  }
