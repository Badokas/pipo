241  int frames(double time, double weight, PiPoValue *values, unsigned int size, unsigned int num)
242  {
243  double onsetThreshold = this->threshold.get();
244  double minimumInterval = this->mininter.get();
245  double durationThreshold = this->durthresh.get();
246  double offThreshold = this->offthresh.get();
247  int colindex = this->colindex.get();
248  int numcols = this->numcols.get();
249  //printf("frames at %f size %d num %d\n", time, size, num);
250 
251  if(size > this->buffer.width)
252  size = this->buffer.width; //FIXME: values += size at the end of the loop can be wrong
253 
254  // clip colindex/size
255  //TODO: this shouldn't change at runtime, so do this in streamAttributes only
256  while (colindex < 0 && size > 0)
257  colindex += size;
258 
259  if (numcols <= 0)
260  numcols = size;
261 
262  if (colindex + numcols > size)
263  numcols = size - colindex;
264 
265  for(unsigned int i = 0; i < num; i++)
266  { // for all frames
267  PiPoValue scale = 1.0;
268  double odf = 0.0;
269  double energy = 0.0;
270 
271  /* normalize sum to one for Kullback Leibler divergence */
272  if(this->onsetMode == KullbackLeiblerOnset)
273  {
274  PiPoValue normSum = 0.0;
275 
276  for(int j = 0; j < numcols; j++)
277  normSum += values[colindex + j];
278 
279  scale = 1.0 / normSum;
280  }
281 
282  /* input frame */
283  int filterSize = this->buffer.input(values, size, scale);
284  this->temp = this->buffer.vector;
285 
286  switch(this->onsetMode)
287  {
288  case MeanOnset:
289  {
290  unsigned int k = colindex;
291  for(int j = 0; j < numcols; j++, k++)
292  {
293  odf += (values[k] - this->lastFrame[k]);
294  energy += values[k];
295 
296  this->lastFrame[k] = rta_selection_stride(&this->temp[k], size, filterSize, (filterSize - 1) * 0.5);
297  }
298 
299  odf /= numcols;
300  energy /= numcols;
301 
302  break;
303  }
304 
305  case MeanSquareOnset:
306  case RootMeanSquareOnset:
307  {
308  unsigned int k = colindex;
309  for(int j = 0; j < numcols; j++, k++)
310  {
311  double diff = values[k] - this->lastFrame[k];
312 
313  odf += (diff * diff);
314  energy += values[k] * values[k];
315 
316  this->lastFrame[k] = rta_selection_stride(&this->temp[k], size, filterSize, (filterSize - 1) * 0.5);
317  }
318 
319  odf /= numcols;
320  energy /= numcols;
321 
322  if(this->onsetMode == RootMeanSquareOnset)
323  {
324  odf = sqrt(odf);
325  energy = sqrt(energy);
326  }
327 
328  break;
329  }
330 
331  case KullbackLeiblerOnset:
332  {
333  unsigned int k = colindex;
334  for(int j = 0; j < numcols; j++, k++)
335  {
336  if(values[k] != 0.0 && this->lastFrame[k] != 0.0)
337  odf += log(this->lastFrame[k] / values[k]) * this->lastFrame[k];
338 
339  energy += values[k] * values[k];
340 
341  this->lastFrame[k] = rta_selection_stride(&this->temp[k], size, filterSize, (filterSize - 1) * 0.5);
342  }
343 
344  odf /= numcols;
345  energy /= numcols;
346 
347  break;
348  }
349  }
350 
351  /* get onset */
352  double maxsize = maxsegsize.get();
353  bool frameIsOnset = (odf > onsetThreshold && !this->lastFrameWasOnset && time >= this->onsetTime + minimumInterval)
354  || (maxsize > 0 && (time >= this->onsetTime + maxsize)); // chop unconditionally after maxsize if given
355  int ret = 0;
356 
357  if(!this->enDuration)
358  {
359  if (!this->odfoutput.get())
360  { /* output marker */
361  if(frameIsOnset)
362  {
363  /* report immediate onset */
364  ret = this->propagateFrames(this->offset + time, weight, NULL, 0, 1);
365  this->onsetTime = time;
366  }
367  }
368  else
369  { /* output odf for each frame*/
370  PiPoValue odfval = odf;
371  ret = this->propagateFrames(this->offset + time, weight, &odfval, 1, 1);
372  }
373  }
374  else
375  {
376  double duration = time - this->onsetTime;
377 
378  if(this->segIsOn && (frameIsOnset || energy < offThreshold) && duration >= durationThreshold)
379  {
380  /* end of segment (new onset or energy below off threshold) */
381  int outputSize = static_cast<int>(this->outputValues.size());
382 
383  this->outputValues[0] = duration;
384 
385  /* get temporal modelling */
386  if(outputSize > 1)
387  this->tempMod.getValues(&this->outputValues[1], outputSize - 1, true);
388 
389  /* report segment */
390  ret = this->propagateFrames(this->offset + this->onsetTime, weight, &this->outputValues[0], outputSize, 1);
391  }
392 
393  /* segment on/off (segment has at least one frame) */
394  if(frameIsOnset)
395  {
396  this->segIsOn = true;
397  this->onsetTime = time;
398  }
399  else if(energy < offThreshold)
400  this->segIsOn = false;
401 
402  /* feed temporal modelling */
403  if(this->segIsOn)
404  this->tempMod.input(values, size);
405  }
406 
407  this->lastFrameWasOnset = frameIsOnset;
408 
409  if(ret != 0)
410  return ret;
411 
412  values += size;
413  time += this->frameperiod; // increase time for next input frame (if num > 1)
414  } // end for all frames
415 
416  return 0;
417  }
